134 uint16_t readCurrentFeedbackAverage_mA()
135 {
136 uint32_t mV;
137
138 if (selectedFeedbackChannel >= 4)
139 {
140 mV = currentSenseSelector1.readAveraged_mV();
141 }
142 else
143 {
144 mV = currentSenseSelector2.readAveraged_mV();
145 }
146
147 // mv = mA * 100 ohms
148 // mA = mV / 100
149 // Actual mA is 300 times this due to current mirror factor in chip
150 // ma = mV * 300 / 100
151 // ma = mV * 3
152 mV *= 3;
153
154 return (uint16_t)mV;
155
156 }
